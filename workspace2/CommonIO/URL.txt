five.btnCtr=function()
{
	five.cliWidth=document.documentElement.clientWidth;
	if(five.part4.offsetWidth>=480)
	{
		if(five.change==null)
		{
			five.change=false;
			five.style1();
		}
		if(five.change)
		{
			five.change=false;
			five.part4.className=five.part4.className.replace(new RegExp("\\b"+"part4a"+"\\b\\s*","g"),"");
			for(var i=0;i<five.info.length;i++)
			{
				five.Icons[i].style.width='3px';
				five.Icons[i].style.height='100%';
			}
			five.Icons[five.now].style.width='100%';
			five.style1();
		}
	}
	else
	{
		if(five.change==null)
		{
			five.change=true;
			five.part4.className+=" part4a";
			five.style2();
		}
		if(!five.change)
		{
			five.change=true;
			five.part4.className+=" part4a";
			for(var i=0;i<five.info.length;i++)
			{
				five.Icons[i].style.width='100%';
				five.Icons[i].style.height='3px';
			}
			five.Icons[five.now].style.height='100%';
			five.style2();
		}
	}
}

five.showPic=function(n)
{
	if(five.Imgs[n])
	{
		five.Imgs[n].src=five.Imgs[n].getAttribute('rel');
		five.Imgs[n]=false;
	}
	for(var i=0;i<five.info.length;i++)
	{
		if(five.Pics[i].style.zIndex>1)five.Pics[i].style.zIndex=five.Pics[i].style.zIndex-1;
	}
	var pNow=five.Pics[n];
	pNow.style.zIndex=3;
	pNow.style.opacity=0;
	var times=10;
	(function(){
		pNow.style.opacity=(11-times)/10;
		times--;
		if(times>0) setTimeout(arguments.callee,10)
		else return
	})()
					
}

five.style1=function()
{
	five.Icons[five.now].style.width='100%';

	{
		five.hover(five.Btns[i],i)

five.hover=function(e,n)
{
	e.onmouseover=function()
	{
		if(five.Icons[n].style.width=='100%') return;
		else
		{
			five.showPic(n)
			five.xx=n;
			for(var i=0;i<five.info.length;i++)
			{
				if(i!=n&&five.Icons[i].offsetWidth!=3)
				{
					five.iconShort(five.Icons[i])
				}
			}
			five.iconLong(five.Icons[n],five.xx);
			five.now=n;
		}
	}
}
five.iconLong=function(e,x)
{
	var nowWidth=five.Btns[0].offsetWidth;
	(function(){
		if(five.xx!=x) return;
		else
		{
			e.style.width=e.offsetWidth+15+"px";
			if(e.offsetWidth>nowWidth)
			{
				e.style.width="100%";
			}
			else setTimeout(arguments.callee,10)
		}
	})();
}
five.iconShort=function(e)
{
	var nowWidth=five.Btns[0].offsetWidth;
	(function(){
		var width = e.offsetWidth-10;
		if(width<3){e.style.width = "3px";}
		else
		{
			e.style.width = width + "px";
			setTimeout(arguments.callee,10);
		}
	})();
}
